Create a new profile and assign the config to the container

``
$ lxc profile show ipmi-manager
config: {}
description: ipmitool manager node
devices:
eth0:
name: eth0
nictype: bridged
parent: br1
type: nic
eth1:
name: eth1
nictype: bridged

size: “0”
type: disk
name: ipmi-manager

$ lxc profile remove c1 default
$ lxc profile assign c1 ipmi-manager
$ lxc exec c1 /bin/bash
$ ls /sys/class/net
